class NityarTools {
  constructor() {
    this.tools = [];
    this.cloudStorage = new CloudManager();
    this.init();
  }

  init() {
    this.loadTools();
    this.setupEventListeners();
    console.log('Nityar initialized');
  }

  loadTools() {
    const tools = [
      new FileManager(),
      new TaskTracker(),
      new CodeSnippets(),
      new Calculator(),
      new TimeTracker(),
      new WebTerminal()
    ];
    
    this.tools = tools;
    return tools;
  }

  setupEventListeners() {
    document.addEventListener('DOMContentLoaded', () => {
      this.render();
    });
  }

  render() {
    const container = document.getElementById('tools-container');
    this.tools.forEach(tool => {
      container.appendChild(tool.element);
    });
  }
}

new NityarTools();

About This Project

Nityar began as a personal project to create a centralized hub for the various tools and utilities I build for myself. Running on a Raspberry Pi, it gives me access to my custom applications and files from anywhere.

Everything here is built with efficiency and practicality in mind. These are real tools I use daily, continuously improved based on my own needs and experiences.
